# Install a package
pip install NOME_DO_MODULO

# Upgrade a package
pip --upgrade NOME_DO_MODULO

# Install all dependencies listed in the file
pip install -r ARQUIVO_DEPENDENCIAS.txt

# Install and upgrade all dependencies in the file
pip install --upgrade -r ARQUIVO_MODULOS.txt

# Uninstall a package
pip uninstall NOME_DO_MODULO

# List installed packages
pip list

# List outdated packages
pip list --outdated
