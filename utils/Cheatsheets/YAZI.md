# Navigation
    kj = Move up/down
    lh = Enter hovered dir / Leave current dir into its parent
    KJ = Seek up/down in the preview
    gg/G = move to the top/bottom
    z = cd to a dir via fzf

# Selection
    <Space> = Toggle selection
    v / V = enter visual mode on selection/unset mode
    Ctrl-a = Select all files
    Ctrl-r = Invert selection of all files
    <Esc> = Cancel selection

# File Operations
    o/<Enter> = Open selected files
    O/S+<Enter> = Open selected files interactively
    <TAB> = Show file information
    y/x= Yank(Copy/Cut) selected files
    p/P = paste/override yanked files
    d/D = Delete(Trash/Permanently) selected files
    a = Create a file/dir
    r = Rename
    . = Toggle hidden files visibility
    ; = Run shell command
    : = Run and block while running a shell command

# Copy path
    cc = Copy file path
    cd = Copy dir path
    cf = Copy file name
    cn = Copy only file name(No extension)

# Filter
    f = Filter files

# Find files
    / = Find next file
    ? = Find previous file
    n/N = Go to the next/previous occurrence
    s = Search files by name using fd
    S = Search files by content using ripgrep
    Ctrl-s = Cancel ongoing search

# Multi-tab
    t = Create new tab with CWD
    1,2,...,9 = Switch to N-th tab
    ]/[ = Switch to the next/previous tab
    }/{ = Swap tabs
    Ctrl-c = Close current tab
